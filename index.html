Esta tabla se carga desde Google Sheets usando Python en el navegador.

from pyscript import display, fetch import csv import io # URL CSV de exportación del Google Sheet url = "https://docs.google.com/spreadsheets/d/1H1m1HAU45aJ6zuqNF7XIGVGcLBJBjS2B/export?format=csv&gid=0" async def cargar_tabla(): try: response = await fetch(url) texto = await response.text() reader = csv.reader(io.StringIO(texto)) rows = list(reader) if not rows: display("La hoja está vacía.", target="estado") else: display("Datos cargados correctamente.", target="estado") html = "" for i, row in enumerate(rows): html += "" tag = "th" if i == 0 else "td" for cell in row: html += f"<{tag}>{cell}" html += "" html += "
" display(html, target="tabla") except Exception as e: display(f"Error al leer la hoja: {e}", target="estado") await cargar_tabla()
